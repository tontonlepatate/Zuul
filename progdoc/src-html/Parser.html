001import java.util.StringTokenizer;
002
003/**
004 * This class is part of the "World of Zuul" application. 
005 * "World of Zuul" is a very simple, text based adventure game.  
006 * 
007 * This parser reads user input and tries to interpret it as an "Adventure"
008 * command. Every time it is called it reads a line from the terminal and
009 * tries to interpret the line as a two word command. It returns the command
010 * as an object of class Command.
011 *
012 * The parser has a set of known command words. It checks user input against
013 * the known commands, and if the input is not one of the known commands, it
014 * returns a command object that is marked as an unknown command.
015 * 
016 * @author  Michael Kolling and David J. Barnes + D.Bureau
017 * @version 2008.03.30 + 2013.09.15
018 */
019public class Parser 
020{
021
022    private CommandWords aCommandWords;  // holds all valid command words
023
024    /**
025     * Create a new Parser.
026     */
027    public Parser() 
028    {
029        this.aCommandWords = new CommandWords();
030    } // Parser()
031
032    /**
033     * Get a new command from the user. The command is read by
034     * parsing the 'inputLine'.
035     */
036    public Command getCommand( final String pInputLine ) 
037    {
038        String vWord1;
039        String vWord2;
040
041        StringTokenizer tokenizer = new StringTokenizer( pInputLine );
042
043        if ( tokenizer.hasMoreTokens() )
044            vWord1 = tokenizer.nextToken();      // get first word
045        else
046            vWord1 = null;
047
048        if ( tokenizer.hasMoreTokens() )
049            vWord2 = tokenizer.nextToken();      // get second word
050        else
051            vWord2 = null;
052
053        // note: we just ignore the rest of the input line.
054
055        // Now check whether this word is known. If so, create a command
056        // with it. If not, create a "null" command (for unknown command).
057
058        if ( this.aCommandWords.isCommand( vWord1 ) )
059            return new Command( vWord1, vWord2 );
060        else
061            return new Command( null, vWord2 );
062    } // getCommand(.)
063
064    /**
065     * Returns a String with valid command words.
066     */
067    public String showCommands() // bad name for this method !
068    {
069        return this.aCommandWords.getCommandList();
070    } // showCommands() // bad name for this method !
071
072} // Parser
