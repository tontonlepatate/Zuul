001import java.util.HashMap;
002import java.util.Set;
003import java.util.Iterator;
004
005public class Room
006{    
007 private String aDescription;
008 private HashMap<String,Room> aExits;
009 private String imageName;
010 private HashMap<String,Item > aItem;
011
012 public Room(final String pDescription,final String pImage)
013 {
014 this.aDescription=pDescription; 
015 aExits = new HashMap<String, Room>();
016 imageName = pImage;
017 aItem = new HashMap<String,Item >();
018 
019 }
020 public String getDescription()
021 {
022 return this.aDescription;    
023 } 
024 public void setExit (final String pDirection,final Room pNeighbor)
025 {
026     aExits.put(pDirection,pNeighbor);
027 }
028
029 public Room getExit(final String pDirection) 
030    {
031        return aExits.get(pDirection);
032    }
033 public String getExitString()
034 {
035        String returnString = "Sorties:";
036        Set<String> keys = aExits.keySet();
037        for(String exit : keys)
038        {
039            returnString +=" "+exit;
040        }
041        return returnString;
042 }
043 public String getLongDescription()
044 {
045     return" Vous etes " + this.aDescription + ".\n" +this.getExitString();
046 }
047 public String getImageName()
048        {
049                return imageName;
050        }
051 public void setItem (final String pName,final Item pItem)
052 {
053     this.aItem.put(pName,pItem);
054 }
055}
