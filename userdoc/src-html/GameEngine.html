131    /**
132     * Given a command, process (that is: execute) the command.
133     * If this command ends the game, true is returned, otherwise false is
134     * returned.
135     */
136    public void interpretCommand(String commandLine) 
137    {
138        gui.println(commandLine);
139        Command command = aParser.getCommand(commandLine);
140
141        if(command.isUnknown()) {
142            gui.println("I don't know what you mean...");
143            return;
144        }
145
146        String commandWord = command.getCommandWord();
147        if (commandWord.equals("help"))
148            printHelp();
149        else if (commandWord.equals("go"))
150            goRoom(command);
151        else if (commandWord.equals("quit")) {
152            if(command.hasSecondWord())
153                gui.println("Quit what?");
154            else
155                endGame();
156        }
157        else if (commandWord.equals("back"))this.back();
158        
159    }

178    /** 
179     * Try to go to one direction. If there is an exit, enter the new
180     * room, otherwise print an error message.
181     */
182    private void goRoom(Command command) 
183    {
184        if(!command.hasSecondWord()) {
185            // if there is no second word, we don't know where to go...
186            gui.println("Go where?");
187            return;
188        }
189
190        String direction = command.getSecondWord();
191
192        // Try to leave current room.
193        Room nextRoom = aCurrentRoom.getExit(direction);
194
195        if (nextRoom == null)
196            gui.println("There is no door!");
197        else {
198            this.aPile2Room.push(aCurrentRoom);
199            aCurrentRoom = nextRoom;
200            gui.println(aCurrentRoom.getLongDescription());
201            if(aCurrentRoom.getImageName() != null)
202                gui.showImage(aCurrentRoom.getImageName());
203        }
204    }
